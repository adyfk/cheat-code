## Method

### String

```js
  let text = 'ABCDEF ';

  // String Getting --------------
  text.length             // = '7'
  // slice(start, end)
  text.slice(2,3)         // = 'C' 
  // substring(start, end)
  text.substring(2,3)     // = 'C'
  // substr(start, length)
  text.substr(2,1)        // = 'C'
  // replace(string | RegExp, string)
  text.replace('A', 'B')  // = 'BBCDEF'
  text.replace(/^A/, 'B') // = 'BBCDEF'
  // toUpperCase & toLowerCase
  text.toLowerCase()      // = 'abcdef'
  text.trim()             // = 'ABCDEF'
  text.padStart(2,1)      // = '11ABCDEF'
  text.padEnd(2,1)        // = 'ABCDEF11'
  text.split('')          // = ['A','B','C','D','E','F']

  // String Search --------------
  text.indexOf('B')       // = 1
  text.match(/F/)         // = ["F",index: 5]
  // The search() method cannot take a second start position argument.
  // lastIndexOf()
  // startsWith()
  // endsWith()
```

### Array

```js
  const fruits = ["Banana", "Orange", "Apple", "Mango"];
  
  // Doesn't effetct to data --------------------------
  fruits.toString()       // = 'Banana,Orange,Apple,Mango'
  fruits.join("*")        // = 'Banana*Orange*Apple*Mango'
  //- fruits.slice(start, end) 
  fruits.slice(1)         // = ["Orange", "Apple", "Mango"]

  // Method will effect to data -----------------------
  fruits.pop();           // = 'Mango'
  // fruits = ['Banana', 'Orange', 'Apple']
  //- fruits.push(...items);
  fruits.push("Kiwi");    // = 4;
  // fruits = ['Banana', 'Orange', 'Apple', 'Kiwi']       
  fruits.shift();         // = 'Banana'
  // fruits = ['Orange', 'Apple', 'Kiwi'] 
  //- fruits.unshift(...items);
  fruits.unshift("Lemon");// = 4
  // fruits = ['Lemon', 'Orange', 'Apple', 'Kiwi']
  //- fruits.splice(start,deleteCount?, ...itemsReplace)
  fruits.splice(1, 1);    // = 'Orange'
  // fruits = ['Lemon', 'Apple', 'Kiwi']
  fruits.sort()           // = ['Apple', 'Kiwi', 'Lemon']
  fruits.reverse()        // = ['Lemon', 'Kiwi', 'Apple']
  
  // Numeric Sort
  const points = [40, 100, 1, 5, 25, 10];
  // sortby highest
  points.sort((a, b)=>a - b);
```
