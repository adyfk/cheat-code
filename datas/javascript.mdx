```js
  var x = 1;

  let a, b = 2, c = 'C';
```

### Data Types
<note>JavaScript has dynamic types. This means that the same variable can be used to hold different data types:</note>

```js
  let x;           // Now x is undefined
  x = 5;           // Now x is a Number
  x = "John";      // Now x is a String
```
<note>
  <strong>String</strong>
</note>

```js
  let version = '123'
  let carName1 = "version " + version;   // Using double quotes
  let carName2 = 'version ' + version;   // Using single quotes
  let carName3 = `version  ${version}`;   // Using backtick

  // convert to string
  let a = (123).toString()
  let b = 123 + ''
```
<note>
  <strong>Number</strong>
</note>

```js
  let x1 = 34.00;     // Written with decimals
  let x2 = 34;        // Written without decimals
  let y = 123e5;      // 12300000
  let z = 123e-5;     // 0.00123

  // convert to number
  let a = Number('123')   // 123
  let b = +'123'          // 123
  let c = parseInt('123') // 123
  let d = parseFloat('123.4') // 123.4
```
* <strong>toExponential()</strong>	Returns a string, with a number rounded and written using exponential notation.
* <strong>toFixed()</strong>	Returns a string, with a number rounded and written with a specified number of decimals.
* <strong>toPrecision()</strong>	Returns a string, with a number written with a specified length

### Conditional
<note>IF</note>

```js
  let data;

    data = true;
  } else {
    data = false;
  }

  if(true) data = true

```
<note>Switch</note>

```js
  let data;
  switch(data){
    case true: {
      data = true;

    }
    default: {
      data = false;
    }
  }
```

### Looping
<note>For</note>

```js
  for (let i = 0; i < 5; i++) {
    console.log("The number is " + i;

    // break;  will stop iteration
    // continue; will skip iteration
  }
```
<note>For In</note>

```js
  // for (key in object | array)
  // if object will return key , if array will return index;
  const person = {fname:"John", lname:"Doe", age:25}
  for (let key in person) {
    console.log(key) 
  }
  // fname, lname, age
```
<note>For Of</note>

```js
  // for (variable of iterable)

  const cars = ["BMW", "Volvo", "Mini"];
  for (let x of cars) {
    console.log(x)
  }
  // BMW, Volvo, Mini
```
<note>While</note>

```js
  // while (condition) 
  while(true){
    console.log('invinity print')
  }
```
