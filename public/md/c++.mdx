### Class 
```cpp
  class BasicClass {  // The class
    public:
      int x;
  }

  class MyClass {      
    private:   // Private access specifier
      int y;   // Private attribute
    public:             // Access specifier
      int myNum;        // Attribute (int variable)
      string myString;  // Attribute (string variable)
      MyClass() {       // Constructor
        cout << "Hello Cheetsheet!";
      }
      MyClass(int x) {  // Constructor
        cout << x;
      }
      void myMethod1() {  // Method/function defined inside the class
        cout << "Hello Cheetsheet!";
      }
      void setSalary(int yVal) {
        y = yVal;
      }
  };

  // Inheritance
  class MyChildClass: public MyClass {
    public:
      string model = "Mustang";
  };

  // Multiple Inheritance
  class MyChildClass: public MyClass, public BasicClass {
    public:
      string model = "Mustang";
  };

  // Method/function definition outside the class
  void MyClass::myMethod2() {
    cout << "Hello Cheetsheet!";
  }

  int main() {
    MyClass myObj, myObj(10);  // Create an object of MyClass

    // Access attributes and set values
    myObj.myNum = 15; 
    myObj.myString = "Some text";

    // Print attribute values
    cout << myObj.myNum << "\n";
    cout << myObj.myString;
    return 0;
  }
```
