### Looping
<note><b>While</b></note>

```py
  while i < 6:
    print(i)
    i += 1

  while i < 6:
    print(i)
    i += 1
  else:
    print("i is no longer less than 6")

  while i < 6:
    print(i)
    if i == 3: break
    i += 1

  while i < 6:
    i += 1
    if i == 3: continue
    print(i)
```
<note><b>For</b></note>

```py
  for x in range(6):
    print(x)

  for x in "cheatcode":
    if x == "t": continue
    print(x)

  for x in [1,2,3,4]:
    if x == 3: break
    print(x)
  else:
    print("Finally finished!")
```

  # basic function
  def my_func():

  my_func() # Hellp
  
  # function with return 
  def my_func3():
    return 1
  print(my_func3()) # 1

  # params with default value
  def my_func3(fname, lname = 'code'):
    print("Name " + fname + " " + lname)
  my_func3("cheatsheet") # Name cheatsheet code

  # Arbitrary Arguments, *args
  def my_func4(*kids):
    print("The youngest child is " + kids[1])
  my_func4("cheatsheet", "code") # code

  # Arbitrary Keyword Arguments, **kwargs
  def my_func5(**kid):
    print(kid["lname"])
  my_func5(fname = "A", lname = "B") # B
