### Looping
<note><b>While</b></note>

```py
  while i < 6:
    print(i)
    i += 1

  while i < 6:
    print(i)
    i += 1
  else:
    print("i is no longer less than 6")

  while i < 6:
    print(i)
    if i == 3: break
    i += 1

  while i < 6:
    i += 1
    if i == 3: continue
    print(i)
```
<note><b>For</b></note>

```py
  for x in range(6):
    print(x)

  for x in "cheatcode":
    if x == "t": continue
    print(x)

  for x in [1,2,3,4]:
    if x == 3: break
    print(x)
  else:
    print("Finally finished!")
```

### Function
```py
  # basic function
  def my_func():
    print("Hello")
  my_func() # Hellp
  
  # function with return 
  def my_func3():
    return 1
  print(my_func3()) # 1

  # params with default value
  def my_func3(fname, lname = 'code'):
    print("Name " + fname + " " + lname)
  my_func3("cheat") # Name cheat code

  # Arbitrary Arguments, *args
  def my_func4(*kids):
    print("The youngest child is " + kids[1])
  my_func4("cheat", "code") # code

  # Arbitrary Keyword Arguments, **kwargs
  def my_func5(**kid):
    print(kid["lname"])
  my_func5(fname = "A", lname = "B") # B
```

### List 
<note>
  <em>Allow Duplicates, Changeable, Ordered, List Items - Data Types</em>
</note>

```py
  thislist = ["apple", "banana", "cherry", "apple", "cherry"]
  # access -------------------
  thislist[1]        # banana
  thislist[-1]       # cherry
  len(thislist)      # 5
  # slice
  thislist[2:5]      # ['cherry', 'apple', 'cherry']
  if "apple" in thislist:
    print("Yes)

  # change -----------------------------------------------
  thislist[1] = "blackcurrant"
  thislist[1:3] = ["blackcurrant", "watermelon"]
  thislist.insert(2, "watermelon")
  thislist.append("orange")
  thislist.remove("banana")
  thislist.pop(1)             # pop spesific index
  del thislist[0]
  thislist.clear()

  # loop -------------------------------------------------
  for x in thislist:
    print(x)

  # Looping Using List Comprehension ---------------------
  [print(x) for x in thislist]
  # filtering 
  # [expression for item in iterable if condition == True]
  newlist = [x for x in thislist if "a" in x]  
  # ['apple', 'banana', 'apple']
  newlist = [x.upper() for x in thislist]
  # ['APPLE', 'BANANA', 'CHERRY', 'APPLE', 'CHERRY']
  newlist = [x if x != "banana" else "orange" for x in thislist] 
  # ['apple', 'orange', 'cherry', 'apple', 'cherry']

  # sort -------------------------------------------------
  thislist.sort()
  thislist.sort(reverse = True)
  # custom sort
  newlist = [100, 50, 65, 82, 23]
  def myfunc(n):
    return abs(n - 50)
  newlist.sort(key = myfunc) 
  # [50, 65, 23, 82, 100]

  # copy -------------------------------------------------
  newlist = thislist.copy()
  newlist = list(thislist)
  
  # Join -------------------------------------------------
  newlist = newlist + thislist
  newlist.extend(newlist)
```

### Tuple 
<note>
  <em>Allow Duplicates, Unchangeable, Ordered, List Items - Data Types</em>
</note>

```py
  thistuple = ("apple", "banana", "cherry", "banana")
  
  # access like list --------------------------------------
  thistuple[1]
  
  # change ------------------------------------------------
  templist = list(thistuple)
  templist[1] = "kiwi"
  thistuple = tuple(templist)

  # unpack ------------------------------------------------
  (a, b) = thistuple 
  # a = apple , b = banana
  (a, *b) = thistuple
  # a = apple , b = ['cherry', 'banana']

  # loop  -------------------------------------------------
  for x in thistuple:
    print(x)

  # join  -------------------------------------------------
  tuple1 = ("a", "b" , "c")
  tuple2 = (1, 2, 3)
  tuple3 = tuple1 + tuple2

  fruits = ("apple", "banana", "cherry")
  mytuple = fruits * 2
  # 'apple', 'banana', 'cherry', 'apple', 'banana', 'cherry')
```
